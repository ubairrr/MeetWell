Class ContextComposer:
  constructor(private db: Database.Database)

Method: getContext(meetingId: string, coveredUntil: number): ContextWindow

  rollingSegments query:
    SELECT id, meeting_id, speaker_label, channel, timestamp_start, timestamp_end, text
    FROM transcript_segments
    WHERE meeting_id = ? AND timestamp_start > ?
    ORDER BY timestamp_start ASC
    Parameters: [meetingId, coveredUntil]
    Cast result to TranscriptSegmentRow[]

  epochSummaries query:
    SELECT id, meeting_id, covered_interval_start, covered_interval_end, decisions_json, action_items_json, key_points_json, speaker_attributions_json, raw_segment_count, token_count_compressed, created_at
    FROM epoch_summaries
    WHERE meeting_id = ?
    ORDER BY covered_interval_start ASC
    Parameters: [meetingId]
    Map each raw row to StoredEpochSummary by parsing JSON fields:
      decisions: JSON.parse(row.decisions_json)
      action_items: JSON.parse(row.action_items_json)
      key_points: JSON.parse(row.key_points_json)
      speaker_attributions: JSON.parse(row.speaker_attributions_json)

  return { rollingSegments, epochSummaries }

Create src/main/context/ContextEngine.ts.

Imports:
- Database from 'better-sqlite3-multiple-ciphers'
- BrowserWindow from 'electron'
- SummaryCardStore from '../store/SummaryCardStore'
- LLMAdapter from '../llm/LLMAdapter'
- EmbeddingAdapter from '../llm/EmbeddingAdapter'
- SummaryCardTimer from './SummaryCardTimer'
- TokenMonitor from './TokenMonitor'
- EpochCompressor from './EpochCompressor'
- ContextComposer, ContextWindow from './ContextComposer'
- RollingWindow from './RollingWindow'
- StoredEpochSummary from '../../shared/schemas/index'

Export interface ContextEnginePort:
  start(meetingId: string): void
  stop(): void
  getContext(): ContextWindow | null
  onEpochCompressed(cb: (summary: StoredEpochSummary) => void): void

Class ContextEngine implements ContextEnginePort:

Constructor args: (db: Database.Database, win: BrowserWindow, summaryCardStore: SummaryCardStore, llm: LLMAdapter, embedding: EmbeddingAdapter)

Private fields, all created in constructor:
  - summaryCardTimer: SummaryCardTimer — new SummaryCardTimer(db, win, summaryCardStore, llm)
  - tokenMonitor: TokenMonitor — new TokenMonitor(db)
  - epochCompressor: EpochCompressor — new EpochCompressor(db, llm, embedding)
  - contextComposer: ContextComposer — new ContextComposer(db)
  - rollingWindow: RollingWindow — new RollingWindow()
  - currentMeetingId: string | null = null
  - epochCallbacks: Array&lt;(summary: StoredEpochSummary) => void&gt; = []

Method start(meetingId: string): void
  1. If currentMeetingId is already set (previous session not stopped), call stop() first.
  2. currentMeetingId = meetingId
  3. summaryCardTimer.start(meetingId)
  4. tokenMonitor.start(meetingId, rollingWindow, async (count) => {
       try {
         const summary = await epochCompressor.compress(meetingId, count, rollingWindow)
         if (summary) {
           for (const cb of epochCallbacks) cb(summary)
         }
       } catch (err) {
         console.error('[ContextEngine] epoch compression error:', err)
       }
     })

Method stop(): void
  1. summaryCardTimer.stop()
  2. tokenMonitor.stop()
  3. rollingWindow.reset()
  4. currentMeetingId = null

Method getContext(): ContextWindow | null
  If currentMeetingId is null, return null.
  return contextComposer.getContext(currentMeetingId, rollingWindow.getCoveredUntil())

Method onEpochCompressed(cb: (summary: StoredEpochSummary) => void): void
  epochCallbacks.push(cb)

SummaryCardTimer is NOT retrofitted — it continues to query transcript_segments directly for its 5-minute window (D-08). ContextEngine only manages its start/stop lifecycle.

Export ContextEngine as named export, ContextEnginePort as named export.
