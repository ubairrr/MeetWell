   - mockEmbedding: object with embed: vi.fn().mockResolvedValue(new Float32Array(1536).fill(0.01))

5. Seed helper: function seedSegments(db, meetingId, count):
   - Insert `count` rows into transcript_segments
   - Each row: id=crypto.randomUUID(), meeting_id, speaker_label='Speaker 0', channel='mic', timestamp_start=i (seconds), timestamp_end=i+1, text=a string long enough to produce approximately 600 tokens when encoded by tiktoken cl100k_base
   - To guarantee &gt;560K tokens with 1000 segments: use a text field of at least 2400 chars (roughly 600 tokens at ~4 chars/token). Example: repeat a lorem sentence to fill 2400 chars.
   - created_at = Date.now() + i

6. Main test: describe('ContextEngine compression pipeline — simulated 60-minute meeting', () => {

   it('EpochCompressor fires exactly once, writes 1 epoch_summaries row, 1 vec_chunks row, rolling window stays below 800K ceiling', async () => {
     const meetingId = 'test-meeting-60min'
     seedSegments(db, meetingId, 1000)

     const rollingWindow = new RollingWindow()
     const tokenMonitor = new TokenMonitor(db)
     const epochCompressor = new EpochCompressor(db, mockLLM as any, mockEmbedding as any)
     const contextComposer = new ContextComposer(db)

     // Step 1: verify token count exceeds threshold BEFORE compression
     const countBefore = await tokenMonitor.countTokens(meetingId, rollingWindow.getCoveredUntil())
     expect(countBefore).toBeGreaterThan(TOKEN_THRESHOLD)

     // Step 2: compress (D-12a — fires exactly once)
     const compressionSpy = vi.spyOn(epochCompressor, 'compress')
     const epoch = await epochCompressor.compress(meetingId, countBefore, rollingWindow)
     expect(epoch).not.toBeNull()
     expect(compressionSpy).toHaveBeenCalledTimes(1)

     // D-12b: exactly one epoch_summaries row
     const epochRows = db.prepare('SELECT * FROM epoch_summaries WHERE meeting_id = ?').all(meetingId)
     expect(epochRows).toHaveLength(1)

     // D-12c: exactly one vec_chunks row
     const vecRows = db.prepare('SELECT chunk_id FROM vec_chunks WHERE meeting_id = ?').all(meetingId)
     expect(vecRows).toHaveLength(1)

     // Step 3: second count AFTER compression — should be below threshold (D-12a: would not fire again)
     const countAfter = await tokenMonitor.countTokens(meetingId, rollingWindow.getCoveredUntil())
     expect(countAfter).toBeLessThan(TOKEN_THRESHOLD)

     // D-12d: rolling window stays below 800K ceiling
     expect(countAfter).toBeLessThan(800_000)

     // Verify rollingWindow watermark advanced (compress called markEvicted)
     expect(rollingWindow.getCoveredUntil()).toBeGreaterThan(0)

     // Verify ContextComposer returns ContextWindow with correct shape after compression
     const ctx = contextComposer.getContext(meetingId, rollingWindow.getCoveredUntil())
     expect(ctx.epochSummaries).toHaveLength(1)
     expect(ctx.epochSummaries[0].decisions).toEqual(['Mock decision'])
     expect(ctx.epochSummaries[0].speaker_attributions).toEqual({ 'Speaker 0': 'led discussion' })
     // rollingSegments should contain only segments after the watermark (evicted segments excluded)
     expect(ctx.rollingSegments.length).toBeLessThan(1000)
   })
 })

Do not use fake timers (vi.useFakeTimers) — the test controls compression directly via method calls, bypassing the setInterval.
